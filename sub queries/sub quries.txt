# query to find jones manager

select *
from emp
where empno in (select mgr
                from emp
                where ename = 'jones');


# query to find dname of jones
select dname
from dept
where deptno = (select deptno
                from emp
                where ename = 'jones');




select avg(sal)
from emp
where deptno = (select deptno
                from dept
                where loc = 'dallas');
